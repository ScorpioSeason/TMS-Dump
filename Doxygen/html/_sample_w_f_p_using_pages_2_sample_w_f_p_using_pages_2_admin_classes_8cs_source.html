209  public TMSLog(string nUnparsed)
210  {
211  // Used in Logger / LogIt call
212  if ((Regex.Matches(nUnparsed, "\\|").Count) == 6)
213  {
214  string[] temp = nUnparsed.Split('|');
215  logPath = temp[1];
216  logClass = temp[2];
217  logMethod = temp[3];
218  logType = temp[4];
219  logMessage = temp[5];
220  logTime = DateTime.UtcNow;
221  }
222  // Used in reading in from a file
223  else if ((Regex.Matches(nUnparsed, "\\|").Count) == 7)
224  {
225  string[] temp = nUnparsed.Split('|');
226  logTime = DateTime.Parse(temp[1]);
227  logPath = temp[2];
228  logClass = temp[3];
229  logMethod = temp[4];
230  logType = temp[5];
231  logMessage = temp[6];
232  }
233  // Used in formatting error
234  else
235  {
236  logPath = "AdminClasses.cs";
237  logClass = "TMSLog";
238  logMethod = "Constructor";
239  logType = "LogParseError";
240  logMessage = "The log message did not enter as the correct string format";
241  logTime = DateTime.UtcNow;
242  nUnparsed = "|" + logPath + "|" + logClass + "|" + logMethod + "|" + logType + "|" + logMessage + "|";
243  }
